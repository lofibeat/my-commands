```bash

✏️ Nano — редактор в терминале (Linux / macOS)

# На macOS nano часто не предустановлен: brew install nano

## Запуск

nano file                      # открыть файл (создаётся, если нет)
nano +line,column file         # открыть с курсором на строке (и столбце)
nano -v file                   # режим просмотра (read-only)
nano -m file                   # включить мышь (переход по клику)

## Сохранение и выход

Ctrl+O                         # записать в файл (WriteOut), Enter подтвердить
Ctrl+X                         # выйти (спросит сохранить, если были изменения)
Ctrl+S                         # сохранить (в части сборок nano)

## Навигация

Ctrl+A                         # в начало строки
Ctrl+E                         # в конец строки
Alt+\                          # в начало файла (Meta+\)
Alt+/                          # в конец файла
Ctrl+Y                         # страница вверх
Ctrl+V                         # страница вниз
Ctrl+_                         # перейти на строку и столбец (Go to line)
Ctrl+C                         # показать текущую позицию (строка, столбец)

## Редактирование

Ctrl+K                         # вырезать строку (в буфер)
Ctrl+U                         # вставить из буфера (Paste)
Ctrl+W                         # поиск (Where is)
Alt+W                          # поиск следующего вхождения
Ctrl+\                         # поиск и замена (Replace)
Alt+T                          # вырезать до конца строки
Ctrl+6                         # начать выделение (Mark), затем Ctrl+K вырезать
Ctrl+D                         # удалить символ под курсором (Delete)
Backspace                      # удалить символ слева

## Блоки и буфер

Alt+A                          # начать выделение (Mark Set)
Ctrl+K                         # вырезать выделенное (или строку)
Ctrl+U                         # вставить
Alt+6                          # копировать в буфер (без вырезания)

## Справка и прочее

Ctrl+G                         # справка (Help)
Ctrl+J                         # выровнять абзац (Justify)
Ctrl+T                         # проверить орфографию (если включено)
Ctrl+R                         # вставить содержимое другого файла (Read File)
Ctrl+W затем Alt+R             # заменить по всему файлу (Replace)
Alt+3                          # показать/скрыть номера строк

## Обозначения в подсказках

^  — Ctrl (например ^O = Ctrl+O)
M  — Alt / Meta (например M-\ = Alt+\)
```
