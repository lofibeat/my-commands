## grep ‚Äî –ø–æ–∏—Å–∫ –ø–æ —Ç–µ–∫—Å—Ç—É

grep pattern file               # —Å—Ç—Ä–æ–∫–∏, —Å–æ–¥–µ—Ä–∂–∞—â–∏–µ pattern, –≤ file
grep pattern file1 file2        # –ø–æ–∏—Å–∫ –≤ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö —Ñ–∞–π–ª–∞—Ö
grep -r pattern dir/            # —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ –ø–æ –∫–∞—Ç–∞–ª–æ–≥—É

grep -n pattern file            # –ø–æ–∫–∞–∑–∞—Ç—å –Ω–æ–º–µ—Ä–∞ —Å—Ç—Ä–æ–∫
grep -c pattern file            # —Ç–æ–ª—å–∫–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π (count)
grep -l pattern file1 file2     # —Ç–æ–ª—å–∫–æ –∏–º–µ–Ω–∞ —Ñ–∞–π–ª–æ–≤, –≥–¥–µ –µ—Å—Ç—å —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ

grep -w pattern file            # —Ü–µ–ª–æ–µ —Å–ª–æ–≤–æ (word), –Ω–µ –ø–æ–¥—Å—Ç—Ä–æ–∫–∞
grep -E "pat1|pat2" file       # —Ä–∞—Å—à–∏—Ä–µ–Ω–Ω—ã–π regex (OR –∏ –¥—Ä.)
grep -F "literal" file         # –±—É–∫–≤–∞–ª—å–Ω–∞—è —Å—Ç—Ä–æ–∫–∞ (fixed), –±–µ–∑ —Å–ø–µ—Ü—Å–∏–º–≤–æ–ª–æ–≤

grep -B 2 pattern file         # 2 —Å—Ç—Ä–æ–∫–∏ –ø–µ—Ä–µ–¥ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ–º (before)

grep -h pattern file1 file2    # –Ω–µ –≤—ã–≤–æ–¥–∏—Ç—å –∏–º—è —Ñ–∞–π–ª–∞ (—É–¥–æ–±–Ω–æ —Å -r)
grep --color=auto pattern file # –ø–æ–¥—Å–≤–µ—Ç–∫–∞ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–π –≤ —Ç–µ—Ä–º–∏–Ω–∞–ª–µ

# –ß–∞—Å—Ç–æ —Å pipe: command | grep pattern

## find ‚Äî –ø–æ–∏—Å–∫ —Ñ–∞–π–ª–æ–≤ –ø–æ –∏–º–µ–Ω–∏ –∏ –∞—Ç—Ä–∏–±—É—Ç–∞–º

find dir -name "*.txt"          # —Ñ–∞–π–ª—ã –ø–æ –∏–º–µ–Ω–∏ (—Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ)
find . -name "*.log"            # –≤ —Ç–µ–∫—É—â–µ–º –∫–∞—Ç–∞–ª–æ–≥–µ –∏ –Ω–∏–∂–µ
find . -iname "*.TXT"           # –±–µ–∑ —É—á—ë—Ç–∞ —Ä–µ–≥–∏—Å—Ç—Ä–∞
find . -type f                  # —Ç–æ–ª—å–∫–æ –æ–±—ã—á–Ω—ã–µ —Ñ–∞–π–ª—ã (f file, d directory, l link)
find . -type d -name "src"      # –∫–∞—Ç–∞–ª–æ–≥–∏ —Å –∏–º–µ–Ω–µ–º src
find . -mtime -7                # –∏–∑–º–µ–Ω—ë–Ω–Ω—ã–µ –∑–∞ –ø–æ—Å–ª–µ–¥–Ω–∏–µ 7 –¥–Ω–µ–π
find . -mtime +30               # –Ω–µ –º–µ–Ω—è–≤—à–∏–µ—Å—è –±–æ–ª—å—à–µ 30 –¥–Ω–µ–π

find . -empty                   # –ø—É—Å—Ç—ã–µ —Ñ–∞–π–ª—ã –∏ –∫–∞—Ç–∞–ª–æ–≥–∏
find . -perm 644                # —Å –ø—Ä–∞–≤–∞–º–∏ 644
find . -user root               # –≤–ª–∞–¥–µ–ª–µ—Ü root
find . -exec rm {} \;           # –≤—ã–ø–æ–ª–Ω–∏—Ç—å –∫–æ–º–∞–Ω–¥—É –¥–ª—è –∫–∞–∂–¥–æ–≥–æ (–æ—Å—Ç–æ—Ä–æ–∂–Ω–æ —Å rm)

find . -delete                  # —É–¥–∞–ª–∏—Ç—å –Ω–∞–π–¥–µ–Ω–Ω–æ–µ (–æ—Å—Ç–æ—Ä–æ–∂–Ω–æ)

## find + xargs (–ø–µ—Ä–µ–¥–∞—á–∞ —Å–ø–∏—Å–∫–∞ –≤ –∫–æ–º–∞–Ω–¥—É)
